## 目录
~~~
├─ framework
│  ├─ config // 配置驱动
│     ├─ env // env配置（全局配置，优先级最高）
│        ├── ...
│     ├─ ini // ini配置（项目独立配置）
│        ├── ...
│     ├─ toml // toml配置（全局配置，优先级低于env）
│        ├── ...
│     ├── ...
│  ├── db // 数据库驱动
│     ├─ ip // IP地址解析库
│        ├── ...
│     ├─ mysql // mysql数据库驱动
│        ├── ...
│     ├─ redis // redis数据库驱动
│        ├── ...
│     ├── ...
│  ├─ logs // 日志驱动
│     ├── ...
│  ├─ middleware // 中间件
│     ├── ...
│  ├─ router // 路由
│     ├── ...
│  ├─ util // util工具
│     ├─ crypto // 常用加密工具
│        ├── ...
│     ├─ filepath // 文件或者文件夹操作工具
│        ├── ...
│     ├── format.go // 数据格式化
│     ├── validate.go // 数据验证
│     ├── ...
│  ├─ validator // 验证器
│     ├── ...
│  ├─ app.go // 框架入口文件
~~~
